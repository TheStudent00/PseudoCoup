#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <stdbool.h>

typedef struct SpaceStation {
    char* name;
    int capacity;
    List* modules;
    Dict* resource_levels;
} SpaceStation;

void SpaceStation___init__(SpaceStation* self, char* name, int capacity)
{
    self->name = name;
    self->capacity = capacity;
    self->modules = (List){"Command", "Life Support", "Laboratory"};
    self->resource_levels = (Dict){{"Oxygen", 95}, {"Power", 100}};
}

void* SpaceStation_printStatus(SpaceStation* self)
{
    printf("Station: " + self->name);
    printf(str("Capacity: ") + str(self->capacity));
}

typedef struct Astronaut {
    char* name;
    int experience;
    SpaceStation* assigned_station;
    int health;
} Astronaut;

void Astronaut___init__(Astronaut* self, char* name, int experience, SpaceStation* assigned_station)
{
    self->name = name;
    self->experience = experience;
    self->assigned_station = assigned_station;
    self->health = 100;
}

int Astronaut_work(Astronaut* self, int hours)
{
    if (hours < 0)
    {
        raise_error(Exception_new("Cannot work negative hours"));
    }
    int tasks_done = 0;
    int current_hour = 0;
    while (current_hour < hours)
    {
        if (self->health < 20)
        {
            printf("Astronaut is too tired to work.");
            break;
        }
        if (current_hour % 3 == 0)
        {
            current_hour = current_hour + 1;
            continue;
        }
        self->health = self->health - 10;
        tasks_done = tasks_done + 2;
        current_hour = current_hour + 1;
    }
    return tasks_done;
}

void* Astronaut_report(Astronaut* self)
{
    if (self->assigned_station != NULL)
    {
        self->assigned_station->printStatus();
        if (List_length(self->assigned_station->modules) > 0)
        {
            printf(str("Primary module: ") + str(self->assigned_station->modules[0]));
        }
        if (Dict_contains(self->assigned_station->resource_levels, "Oxygen"))
        {
            int oxy = self->assigned_station->resource_levels["Oxygen"];
            printf(str("Oxygen Level: ") + str(oxy));
        }
    }
    else
    {
        printf("No station assigned.");
    }
}

        Astronaut* astro = Astronaut_new("Ripley", 10, station);
        astro->report();
        int completed = astro->work(8);
        printf(str("Tasks Completed: ") + str(completed));
        printf(str("Health remaining: ") + str(astro->health));
        Astronaut* rookie = Astronaut_new("Newb", 0, NULL);
        rookie->report();
        rookie->work(-5);
    }
    else if (Pseudo_CATCH(&e))
    {
        printf("Caught expected exception in Space Station");
    }
    return 0;
}
